5. api/handlers.rs

Contains the main API implementation.

rust
use poem_openapi::{param::Path, payload::Json, OpenApi};
use tokio::sync::Mutex;
use slab::Slab;

use super::types::{User, UpdateUser, ApiTags};
use super::responses::*;

#[derive(Default)]
pub struct Api {
    users: Mutex<Slab<User>>,
}

#[OpenApi]
impl Api {
    #[oai(path = "/users", method = "post", tag = "ApiTags::User")]
    async fn create_user(&self, user: Json<User>) -> CreateUserResponse {
        let mut users = self.users.lock().await;
        let id = users.insert(user.0) as i64;
        CreateUserResponse::Ok(Json(id))
    }

    #[oai(path = "/users/:user_id", method = "get", tag = "ApiTags::User")]
    async fn find_user(&self, user_id: Path<i64>) -> FindUserResponse {
        let users = self.users.lock().await;
        match users.get(user_id.0 as usize) {
            Some(user) => FindUserResponse::Ok(Json(user.clone())),
            None => FindUserResponse::NotFound,
        }
    }

    #[oai(path = "/users/:user_id", method = "delete", tag = "ApiTags::User")]
    async fn delete_user(&self, user_id: Path<i64>) -> DeleteUserResponse {
        let mut users = self.users.lock().await;
        if users.contains(user_id.0 as usize) {
            users.remove(user_id.0 as usize);
            DeleteUserResponse::Ok
        } else {
            DeleteUserResponse::NotFound
        }
    }

    #[oai(path = "/users/:user_id", method = "put", tag = "ApiTags::User")]
    async fn put_user(&self, user_id: Path<i64>, update: Json<UpdateUser>) -> UpdateUserResponse {
        let mut users = self.users.lock().await;
        match users.get_mut(user_id.0 as usize) {
            Some(user) => {
                if let Some(name) = update.0.name {
                    user.name = name;
                }
                if let Some(password) = update.0.password {
                    user.password = password;
                }
                UpdateUserResponse::Ok
            }
            None => UpdateUserResponse::NotFound,
        }
    }
}

use poem_openapi::{param::Path, payload::Json, OpenApi};
use sqlx::PgPool;

use crate::db::user_repo;
use super::{types::{CreateUser, UpdateUser, User, ApiTags}, responses::*};

pub struct Api {
    db: PgPool,
}

impl Api {
    pub fn new(db: PgPool) -> Self {
        Self { db }
    }
}

#[OpenApi]
impl Api {
    #[oai(path = "/users", method = "post", tag = "ApiTags::User")]
    async fn create_user(&self, user: Json<CreateUser>) -> CreateUserResponse {
        match user_repo::create_user(&self.db, &user.0).await {
            Ok(id) => CreateUserResponse::Ok(Json(id)),
            Err(sqlx::Error::Database(e)) if e.message().contains("duplicate") => CreateUserResponse::Conflict,
            Err(_) => CreateUserResponse::InternalError,
        }
    }

    #[oai(path = "/users/:user_id", method = "get", tag = "ApiTags::User")]
    async fn find_user(&self, user_id: Path<i64>) -> FindUserResponse {
        match user_repo::find_user(&self.db, user_id.0).await {
            Ok(Some(user)) => FindUserResponse::Ok(Json(user)),
            Ok(None) => FindUserResponse::NotFound,
            Err(_) => FindUserResponse::NotFound,
        }
    }

    #[oai(path = "/users/:user_id", method = "delete", tag = "ApiTags::User")]
    async fn delete_user(&self, user_id: Path<i64>) -> DeleteUserResponse {
        match user_repo::delete_user(&self.db, user_id.0).await {
            Ok(true) => DeleteUserResponse::Ok,
            Ok(false) => DeleteUserResponse::NotFound,
            Err(_) => DeleteUserResponse::NotFound,
        }
    }

    #[oai(path = "/users/:user_id", method = "put", tag = "ApiTags::User")]
    async fn put_user(&self, user_id: Path<i64>, update: Json<UpdateUser>) -> UpdateUserResponse {
        match user_repo::update_user(&self.db, user_id.0, &update.0).await {
            Ok(true) => UpdateUserResponse::Ok,
            Ok(false) => UpdateUserResponse::NotFound,
            Err(_) => UpdateUserResponse::NotFound,
        }
    }
}

use sqlx::{PgPool, Error};
use crate::api::types::{CreateUser, UpdateUser, User};

pub async fn create_user(pool: &PgPool, new_user: &CreateUser) -> Result<i64, Error> {
    let rec = sqlx::query!(
        "INSERT INTO users (name, password, email) VALUES ($1, $2, $3) RETURNING id",
        new_user.name,
        new_user.password.expose_secret(),
        new_user.email.to_string(),
    )
    .fetch_one(pool)
    .await?;
    Ok(rec.id)
}

pub async fn find_user(pool: &PgPool, id: i64) -> Result<Option<User>, Error> {
    let result = sqlx::query_as!(
        User,
        "SELECT id, name, password, email FROM users WHERE id = $1",
        id
    )
    .fetch_optional(pool)
    .await?;

    Ok(result)
}

pub async fn delete_user(pool: &PgPool, id: i64) -> Result<bool, Error> {
    let result = sqlx::query!("DELETE FROM users WHERE id = $1", id)
        .execute(pool)
        .await?;
    Ok(result.rows_affected() > 0)
}

pub async fn update_user(pool: &PgPool, id: i64, update: &UpdateUser) -> Result<bool, Error> {
    let current = find_user(pool, id).await?;
    if current.is_none() {
        return Ok(false);
    }

    let user = current.unwrap();
    let name = update.name.clone().unwrap_or(user.name);
    let password = update
        .password
        .as_ref()
        .map(|p| p.expose_secret().to_string())
        .unwrap_or(user.password);

    let result = sqlx::query!(
        "UPDATE users SET name = $1, password = $2 WHERE id = $3",
        name,
        password,
        id
    )
    .execute(pool)
    .await?;

    Ok(result.rows_affected() > 0)
}
